def to_h
  ret = {}
  
  @additions.each do |k,v|
    ret[k.to_s] = { 
      action: ADD, 
      value: v
    }
  end
  @deletions.each do |k,v|
    ret[k.to_s] = {
      action: DELETE,
      value: v
    }
  end
  @updates.each do |k,v|
    ret[k.to_s] = {
      action: PUT,
      value: v
    }
  end

  ret
end
